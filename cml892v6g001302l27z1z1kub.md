```js
function reverseWords(s) {
    let result = [];
    let i = s.length - 1;

    while (i >= 0) {
        // Step 1: Skip spaces
        while (i >= 0 && s[i] === ' ') {
            i--;
        }

        if (i < 0) break;

        // Step 2: Capture a word
        let j = i;
        while (j >= 0 && s[j] !== ' ') {
            j--;
        }

        // Step 3: Push word
        result.push(s.substring(j + 1, i + 1));

        // Step 4: Move pointer
        i = j - 1;
    }

    return result.join(" ");
}
```
