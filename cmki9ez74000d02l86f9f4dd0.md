## ğŸ§¾ Code (Simple JS)

```js
let nums = [1, 1, 0, 1, 1, 1];
const findMaxConsecutiveOnes=(nums)=>{

 // handle edge cases
  if (!Array.isArray(nums) || nums.length === 0) {
    return 0;
  }
let count = 0;
let max = 0;
  

for (let i = 0; i < nums.length; i++) {
  if (nums[i] === 1) {
    count++;
    if (count > max) {
      max = count;
    }
  } else {
    count = 0;
  }
}
return max
}

const result =findMaxConsecutiveOnes(nums)
console.log(result); // Output: 3
```
